    class Menu {
        <<struct>>
        -breakfast : string
        -lunch : string
        -dinner : string
    }

        <<struct>>
        -breakfast : string
        -lunch : string
        -dinner : string
        -mealsServed : int

    class WeeklyMenu {
        <<struct>>
        -days : DailyMenu[7]

    class MealBooking {
        <<struct>>
        -studentId : string

        -confirmed : bool
        -served : bool
    }

    class Appointment {
        <<struct>>
        -studentId : string
        -date : string

        -reason : string
        -status : string
    }

    class Enrollment {
        <<struct>>
        -studentId : string
        -status : string

    class ActivityLogItem {
        <<struct>>

    class Person {
        <<abstract>>
        #id : string
        #fullName : string
        +getId() string
        +getFullName() string
        +setFullName(name)
        +role() string*
    }

    class Student {
        -academicYear : int
        -dormitoryId : string
        -roomNumber : string
        +getAcademicYear() int
        +setAcademicYear(y)
        +isAccommodated() bool
        +assignAccommodation(dormId, room)
        +clearAccommodation()
        +accommodationStatus() string
        +role() string
    }

    class Activity {
        <<abstract>>
        #name : string
        #schedule : string
        #enrollments : Enrollment[]
        +apply(studentId)
        +approve(studentId)
        +refuse(studentId)
        +withdraw(studentId)
        +enroll(studentId)
        +unenroll(studentId)
        +hasEnrollment(studentId) bool
        +category() string*
        +describe() string
    }

    class SportsActivity {
        -coach : string
        +getCoach() string
        +category() string
        +describe() string
    }

    class CulturalActivity {
        -venue : string
        +getVenue() string
        +category() string
        +describe() string
    }

    class Room {
        -number : string
        -capacity : int
        -type : string
        -maintenance : bool
        -occupants : string[]
        +getCapacity() int
        +isFull() bool
        +isEmpty() bool
        +addOccupant(studentId)
        +removeOccupant(studentId)
        +status() string
    }

    class Restaurant {
        -name : string
        -todayMenu : Menu
        -mealsServedToday : int
        +getMenu() Menu
        +setMenu(m)
        +serveMeal()
        +resetMeals()
    }

    class Dormitory {
        -id : string
        -name : string
        -rooms : Room[]

        +addRoom(r)
        +findRoom(number) Room
        +removeRoom(number)
        +totalCapacity() int
        +totalOccupancy() int
        +assignStudent(room, studentId)
        +removeStudent(room, studentId)
    }

    class HealthClinic {
        -appointments : Appointment[]
        +schedule(a)
        +cancel(index)
    }

    class University {
        -name : string
        -students : Student[]

        +loadFromFiles(dir)
